pxMouseValue = pmouseX;

yMouseValue = mouseY;
pyMouseValue = mouseY;

xDiff = pxMouseValue - xMouseValue ;

yDiff = pyMouseValue - yMouseValue ;

if (xDiff > 0){
//println("Links nach Rechts");
line(mouseX, 0, mouseX,500);
}

else if(xDiff < 0){
//println("Rechts nach Links");
line(mouseX, 0, mouseX,500);
}

else {
//println("no movement");

}

//y axis line

if (yDiff > 0){
//println("Links nach Rechts");
line(0, mouseY, 500,mouseY);

else if(yDiff < 0){
//println("Rechts nach Links");
line(0, mouseY, 500,mouseY);
}

else {
//println("no movement");
